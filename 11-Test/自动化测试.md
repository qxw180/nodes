## E2E 测试

```JavaScript
function add(x, y) {
    return x + y
}
function minus(x, y) {
    return x - y
}
function multi(x, y) {
    return x + y
}

function expect(value) {
    return {
        toBe: (actual) => {
            if(actual !== value) {
                throw new Error(`预期值与实际值不相等：预期值为${actual}，实际值为${value}`)
            }
        }
    }
}
function test(desc, fun) {
    try {
        fun()
        console.log(`${desc} 通过测试`)
    } catch (e) {
        console.log(`${desc} 未通过测试（${e}）`)
    }
}
test('测试加法： add(3, 7)', () => {
    expect(add(3, 7)).toBe(10)
})
test('测试减法： minus(3, 3)', () => {
    expect(minus(3, 3)).toBe(0)
})
test('测试乘法： minus(3, 3)', () => {
    expect(multi(3, 3)).toBe(9)
})
```
