    url("//at.alicdn.com/t/webfont_t8iadkg2hvr.woff") format("woff"), /* chrome、firefox */

      url("//at.alicdn.com/t/webfont_t8iadkg2hvr.svg#Alibaba-PuHuiTi-Regular")
      format("svg"); /* iOS 4.1- */
}
